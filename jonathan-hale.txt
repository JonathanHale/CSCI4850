# Create Tables:

create table Organization (
  ID int primary key,
  Name varchar(50),
  NumOfTeams int
);
create table Team (
  ID int primary key,
  Name varchar(50),
  OrgID int
);
create table Player (
  ID int key,
  Email varchar(50),
  TeamID int
);
create table OrgStats (
  Rank int,
  CreateDate date,
  OrgID int
);
create table TeamStats (
  Rank int,
  MatchesWon int,
  MatchesLost int,
  TeamID int
);
create table PlayerStats (
  Rank int,
  CreateDate date,
  PlayerID int
);
create table Characters (
  ID int primary key,
  Name varchar(50),
  class varchar(15),
  PlayerID int
);
create table CharacterStats(
  Rank int,
  ShotsFired int,
  Hits int,
  Misses int,
  Kills int,
  Deaths int,
  CharID int
);
create table Game (
  team1ID int,
  Team2ID int,
  WinnerID int
);
create table Created(
  playerID int not null,
  characterID int not null,
  constraint pk_created primary key (playerID,characterID)
);
create table Joins(
  playerID int not null,
  teamID int not null,
  constraint pk_join primary key (playerID,teamID)
);
create table Owns(
  orgID int not null,
  teamID int not null,
  constraint pk_own primary key (orgID,teamID)
);
create table CompeteIn(
  team1ID int not null,
  team2ID int not null,
  constraint pk_compete primary key (team1ID,team2ID)
);
create table OrgGenerates(
  orgID int not null,
  ID int not null,
  constraint pk_orgGen primary key (orgID,ID)
);
create table TeamGenerates(
  teamID int not null,
  ID int not null,
  constraint pk_TeamGen primary key (TeamID,ID)
);
create table PlayerGenerates(
  playerID int not null,
  ID int not null,
  constraint pk_PlayerGen primary key (playerID,ID)
);
create table CharGenerates(
  CharID int not null,
  ID int not null,
  constraint pk_charGen primary key (CharID,ID)
);

# Foreign Keys:

alter table Team add constraint fk_team foreign key (OrgID)
references Organization(ID) on delete cascade on update cascade;
alter table Player add constraint fk_player foreign key (TeamID)
references Team(ID) on delete cascade on update cascade;
alter table OrgStats add constraint fk_stats foreign key (OrgID)
references Organization(ID) on delete cascade on update cascade;
alter table TeamStats add constraint fk_teamstats foreign key (teamID)
references Team(ID) on delete cascade on update cascade;
alter table PlayerStats add constraint fk_playerstats foreign key (PlayerID)
references Player(ID) on delete cascade on update cascade;
alter table Characters add constraint fk_charplayer foreign key (PlayerID)
references Player(ID) on delete cascade on update cascade;
alter table CharacterStats add constraint fk_char foreign key (charID)
references Characters(ID) on delete cascade on update cascade;
alter table Game add constraint fk_game foreign key (team1ID)
references Team(ID) on delete cascade on update cascade;
alter table Game add constraint fk_game2 foreign key (team1ID)
references Team(ID) on delete cascade on update cascade;
alter table Created add constraint fk_created foreign key(playerID)
references Player(ID) on delete cascade on update cascade;
alter table Created add constraint fk_created2 foreign key(characterID)
references Characters(ID) on delete cascade on update cascade;
alter table Joins add constraint fk_join foreign key(playerID)
references Player(ID) on delete cascade on update cascade;
alter table Joins add constraint fk_join2 foreign key(teamID)
references Team(ID) on delete cascade on update cascade;
alter table Owns add constraint fk_own foreign key(orgID)
references Organization(ID) on delete cascade on update cascade;
alter table Owns add constraint fk_own2 foreign key(teamID)
references Team(ID) on delete cascade on update cascade;
alter table CompeteIn add constraint fk_compete foreign key(team1ID)
references Team(ID) on delete cascade on update cascade;
alter table CompeteIn add constraint fk_compete2 foreign key(team2ID)
references Team(ID) on delete cascade on update cascade;
alter table OrgGenerates add constraint fk_orgGen foreign key(orgID)
references Organization(ID) on delete cascade on update cascade;
alter table OrgGenerates add constraint fk_orgGen2 foreign key(ID)
references OrgStats(OrgID) on delete cascade on update cascade;
alter table TeamGenerates add constraint fk_teamGen foreign key(teamID)
references Team(ID) on delete cascade on update cascade;
alter table TeamGenerates add constraint fk_teamGen2 foreign key(ID)
references TeamStats(TeamID) on delete cascade on update cascade;
alter table PlayerGenerates add constraint fk_playerGen foreign key(playerID)
references Player(ID) on delete cascade on update cascade;
alter table PlayerGenerates add constraint fk_playerGen2 foreign key(ID)
references PlayerStats(PlayerID) on delete cascade on update cascade;
alter table CharGenerates add constraint fk_charGen foreign key(CharID)
references Characters(ID) on delete cascade on update cascade;
alter table CharGenerates add constraint fk_charGen2 foreign key(ID)
references CharacterStats(CharID) on delete cascade on update cascade;

select Name from Organization where ID in (select OrgID from Team where ID in (select TeamID from Player where ID = 9));#get the name of the organization(s) that the player with id 9 belongs to.
select max(shotsFired) from CharacterStats; #get the highest number of shots fired by a single character
select Email from Player where ID in (select PlayerID from Characters where ID in (select CharID from CharacterStats where ShotsFired in (select max(shotsFired) from CharacterStats))); #get the email of the player with the character that fired the most shots.
select Rank,avg(ShotsFired) as avg_shots from CharacterStats group by Rank; #Get the average number of shots fired for each character rank.
select WinnerID from Game group by WinnerID having count(WinnerID) > 1; #Get the Ids of the team(s) that have had more than one win
select Organization.Name as Org_Name, Team.Name as Team_Name from Organization inner join Team on Organization.ID = Team.OrgID; #Get the names of the teams and the organizations they belong to
select Name,Class,PlayerID,TeamID from Characters inner join Player on Characters.PlayerID = Player.ID; #Get the Name,Class,PlayerID, and TeamID from joining the Characters table and the Player table.
select * from (select Name,Class,PlayerID,TeamID from Characters inner join Player on Characters.PlayerID = Player.ID) as newTable where class = 'fighter'; #Get only Characters that have the class 'fighter' from the join of the Characters and Player tables.
select distinct Team.Name as Team_Name from (select * from (select Name,Class,PlayerID,TeamID from Characters inner join Player on Characters.PlayerID = Player.ID) as newTable where class = 'fighter') as TeamTable inner join Team on Team.ID = TeamID; #Get the names of all of the teams that have the class 'fighters' on them.
select Team.Name as Team_Name,count(Team.Name) from (select * from (select Name,Class,PlayerID,TeamID from Characters inner join Player on Characters.PlayerID = Player.ID) as newTable where class = 'fighter') as TeamTable inner join Team on Team.ID = TeamID group by Team.Name; #Get the number of 'fighters' on each team that has them.
select Team_Name from (select Team.Name as Team_Name,count(Team.Name) as team_count from (select * from (select Name,Class,PlayerID,TeamID from Characters inner join Player on Characters.PlayerID = Player.ID) as newTable where class = 'fighter') as TeamTable inner join Team on Team.ID = TeamID group by Team.Name having team_count = 3) as FCount; #Get the team names of the teams that have exactly 3 'fighters'.

select Rank,Name as Char_Name,Class,Email from CharacterStats right join (Characters right join Player on Characters.PlayerID = Player.ID) on CharacterStats.CharID = Characters.ID; #Get the Rank, Name, and class of a character, as well as the email of the player the character belongs to using a right outer join.
select Name,Hits,Misses from Characters left join (CharGenerates left join CharacterStats on CharGenerates.ID = CharacterStats.CharID) on Characters.ID = CharacterStats.CharID; #Get all Characters Name, number of hits, and number of misses using a left join with the CharacterStats and Characters tables.
